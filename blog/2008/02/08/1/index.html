把结构体写入到文件

2008-02-08

今天看到网上一个乱七八糟的乱子，整理一下：
#include <stdio.h>   
#include <string.h>   
struct ST     
{  
    int NO;   
    char name[16];   
};   
int main()   
{
    ST test;
    ST read;   
    FILE *pFile;   
    test.NO = 1;   
    strcpy(test.name, “slackcode.cn”);
    pFile = fopen(“file1.dat”, “w+”);
    if ( !pFile )
    {
        printf(“open file error!”);
        return -1;
    }
 
    rewind(pFile);
    fwrite(&test, sizeof(ST), 1, pFile);

    rewind(pFile);
    fread(&read, sizeof(ST), 1, pFile);

    printf(“%d, %s\n”, read.NO, read.name);

    fclose(pFile);
    return 0;
}
